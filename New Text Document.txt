#!/usr/bin/env python3
import json, os, shutil, subprocess
from pathlib import Path
import boto3

ROLE = "arn:aws:iam::811261691581:role/GlueReleasePushRole"
REPO = "https://git-codecommit.us-east-1.amazonaws.com/v1/repos/glue-production"
BRANCH, DEST = "main", Path("/tmp/glue-production")

def run(cmd, cwd=None, env=None, check=True):
    print("$", " ".join(cmd))
    r = subprocess.run(cmd, cwd=cwd, env=env)
    if check and r.returncode:
        raise Exception(f"Command failed: {' '.join(cmd)}")
    return r.returncode

with open("release-jobs.json") as f:
    m = json.load(f)

release = m["release"]
description = m.get("description", "")
jobs = m["jobs"]

for job in jobs:
    if not Path(job).is_dir():
        raise Exception(f"{job} folder not found")
    if not Path(job, f"{job}.py").exists():
        raise Exception(f"{job}.py missing")
    if not Path(job, f"{job}.json").exists():
        raise Exception(f"{job}.json missing")

c = boto3.client("sts").assume_role(
    RoleArn=ROLE,
    RoleSessionName="GlueRelease"
)["Credentials"]

env = os.environ.copy()
env.update({
    "HOME": "/tmp",
    "AWS_DEFAULT_REGION": "us-east-1",
    "AWS_ACCESS_KEY_ID": c["AccessKeyId"],
    "AWS_SECRET_ACCESS_KEY": c["SecretAccessKey"],
    "AWS_SESSION_TOKEN": c["SessionToken"]
})

shutil.rmtree(DEST, ignore_errors=True)

for cmd in [
    ["git","config","--global","credential.helper","!aws codecommit credential-helper $@"],
    ["git","config","--global","credential.UseHttpPath","true"],
    ["git","config","--global","user.name","Glue Release Pipeline"],
    ["git","config","--global","user.email","gluepipeline@gmail.com"],
]:
    run(cmd, env=env)

run(["git","clone",REPO,str(DEST)], env=env)

if run(["git","show-ref","--verify","--quiet",f"refs/remotes/origin/{BRANCH}"],
       cwd=DEST, env=env, check=False) == 0:
    run(["git","checkout","-B",BRANCH,f"origin/{BRANCH}"], cwd=DEST, env=env)
else:
    run(["git","checkout","-B",BRANCH], cwd=DEST, env=env)

for job in jobs:
    shutil.rmtree(DEST/job, ignore_errors=True)
    shutil.copytree(job, DEST/job)

run(["git","add","--all"], cwd=DEST, env=env)

if run(["git","diff","--cached","--quiet"], cwd=DEST, env=env, check=False):
    msg = f"Release {release}" + (f" - {description}" if description else "")
    run(["git","commit","-m",msg], cwd=DEST, env=env)
    run(["git","push","-u","origin",f"HEAD:{BRANCH}"], cwd=DEST, env=env)
else:
    print("No changes to push.")

shutil.rmtree(DEST, ignore_errors=True)
print(f"Release {release} completed successfully.")
